SELECT  

    Market, 

    `Order Region`, 

    `Category Name`, 

    `Shipping Mode`, 

    COUNT(`Order Id`) AS total_orders, 

    -- Metric 1: Average Days Variance 

    ROUND(AVG(`Days for shipping real` - `Days for shipment scheduled`), 
    1) AS avg_days_variance, 

    -- Metric 2: Late Delivery Rate 

    ROUND( 
        COUNT(CASE WHEN `Delivery Status` = 'Late delivery' THEN 1 END) * 100.0 / COUNT(`Order Id`),
        2 ) AS late_delivery_rate_pct, 

    -- Metric 3: Cancellation Rate 

    ROUND( 
      COUNT(CASE WHEN `Order Status` = 'CANCELED' THEN 1 END) * 100.0 / COUNT(`Order Id`),  
        2 ) AS cancellation_rate_pct, 

    -- Metric 4: Financial Performance 

    ROUND(SUM(`Sales`), 2) AS total_sales, 

    ROUND(SUM(`Order Profit Per Order`), 2) AS total_profit, 

    -- Metric 5: Total Direct Losses (Sold at a Loss) 

    ROUND(SUM(CASE WHEN `Order Profit Per Order` < 0 THEN `Order Profit Per Order` ELSE 0 END), 2) AS total_losses 

FROM `analog-provider-484701-b9.project2.2` 
GROUP BY Market, `Order Region`, `Category Name`, `Shipping Mode` 

ORDER BY total_orders DESC;
